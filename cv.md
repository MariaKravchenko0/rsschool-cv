function toBcd(number){
 let binaryNumber = Math.abs(number)
    .toString()
    .split("")
    .map((n) => (n === "-" ? "-" : parseInt(n).toString(2).padStart(4, "0")))
    .join(" ");
  
  return number < 0 ? "-" + binaryNumber : binaryNumber;
}
